  const ratio = (this.value - this.min) / (this.max - this.min) * 100
  this.style.background = `linear-gradient(90deg, #fff ${ratio}%, ${inactiveColor} ${ratio}%)`
}

  const ratio = (this.value - this.min) / (this.max - this.min) * 100

}
